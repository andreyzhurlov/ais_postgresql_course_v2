CREATE SCHEMA IF NOT EXISTS dev;

------------------------------------------------
------------------------------------------------

drop table if exists dev.clients;

------------------------------------------------
------------------------------------------------

CREATE TABLE If NOT EXISTS dev.clients (
    row_timestamp TIMESTAMP DEFAULT CURRENT_TIMESTAMP,
    id INT PRIMARY KEY,
    first_name VARCHAR(50),
    last_name VARCHAR(50),
    birth_date DATE,
    email VARCHAR(100),
    phone VARCHAR(20),
    city VARCHAR(50)
);
