CREATE SCHEMA IF NOT EXISTS store;

------------------------------------------------
------------------------------------------------

-- Удаляем таблицу
drop table if exists store.clients;

------------------------------------------------
------------------------------------------------

CREATE TABLE If NOT EXISTS store.clients (
    row_timestamp TIMESTAMP DEFAULT CURRENT_TIMESTAMP,
    id INT PRIMARY KEY,
    first_name VARCHAR(50),
    last_name VARCHAR(50),
    birth_date DATE,
    email VARCHAR(100),
    phone VARCHAR(20),
    city VARCHAR(50)
);
